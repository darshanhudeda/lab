def linear_search(arr, key):
    for i in range(len(arr)):
        if arr[i] == key:
            return i
    return -1

def binary_search(arr, key):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == key:
            return mid  # Store the position
        elif arr[mid] < key:
            left = mid + 1
        else:
            right = mid - 1
    return -1

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class CircularLinkedList:
    def __init__(self):
        self.head = None
    def insert_at_beginning(self, data):
        newnode = Node(data)
        if self.head is None:
            self.head = newnode
            newnode.next = self.head
        else:
            newnode.next = self.head
            temp = self.head
            while temp.next != self.head:
                temp = temp.next
            temp.next = newnode
            self.head = newnode

    def delete(self, data):
        if self.head is None:
            print("The list is empty.")
            return
        temp = self.head
        prev = None
        # Check if the head node itself holds the data to be deleted
        if temp.data == data:
            if temp.next == self.head:  # Only one element in the list
                self.head = None
            else:
                while temp.next != self.head:
                    temp = temp.next
                temp.next = self.head.next
                self.head = self.head.next
            return

        # Search for the node to be deleted
        while temp.next != self.head and temp.data != data:
            prev = temp
            temp = temp.next

        if temp == self.head:  # If we didn't find the element
            print("Element not found.")
        else:
            prev.next = temp.next

    # Display the list
    def display(self):
        if self.head is None:
            print("The list is empty.")
            return
        temp = self.head
        while True:
            print(temp.data, end=" -> ")
            temp = temp.next
            if temp == self.head:
                break
        print()

4. Write a menu driven program to perform operations on a stack. (linked list implementation)
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None
class Stack:
    def __init__(self):
        self.top = None
    def push(self, data):
        newnode = Node(data)
        newnode.next = self.top
        self.top = newnode
    def pop(self):
        if self.top is None:
            print("Stack underflow! The stack is empty.")
            return None
        popped_data = self.top.data
        self.top = self.top.next
        print(f"{popped_data} popped from the stack.")
    def peek(self):
        if self.top is None:
            print("Stack is empty. Nothing to peek.")
        else:
            print("Peeked Element is:" ,self.top.data)
    def display(self):
        if self.top is None:
            print("The stack is empty.")
            return
        print("Stack elements are:")
        temp = self.top
        while temp is not None:
            print(temp.data, end=" -> ")
            temp = temp.next
        print("None")

6. Write a menu driven program to perform operations on a circular queue (enqueue, dequeue and peek). (linked list implementation)
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None
class CircularQueue:
    def __init__(self):
        self.tail = None
    def enqueue(self, data):
        newnode = Node(data)
        if self.tail is None:
            self.tail = newnode
            self.tail.next = newnode  # Circular link
        else:
            newnode.next = self.tail.next
            self.tail.next = newnode
            self.tail = newnode

    def dequeue(self):
        if self.tail is None:
            return "Queue is empty."
        head = self.tail.next
        if self.tail == head:  # Only one element in the queue
            self.tail = None
        else:
            self.tail.next = head.next
        return head.data
    def peek(self):
        if self.tail is None:
            return "Queue is empty."
        return self.tail.next.data
    def display(self):
        if self.tail is None:
            return "Queue is empty."
        result = []
        current = self.tail.next
        while True:
            result.append(current.data)
            if current == self.tail:
                break
            current = current.next
        return " -> ".join(map(str, result))

def menu():

7. Write a program to sort the array in ascending order using a) Quick sort b)

def partition(array, low, high):
    pivot = array[low]
    i = low + 1
    j = high
    while i <= j:
        while i <= high and array[i] <= pivot:
            i += 1
        while array[j] > pivot:
            j -= 1
        if i < j:
            array[i], array[j] = array[j], array[i]
    array[low], array[j] = array[j], array[low]
    return j

def quicksort(array, low, high):
    if low < high:
        pivot = partition(array, low, high)
        quicksort(array, low, pivot - 1)
        quicksort(array, pivot + 1, high)


def mergesort(arr):
    if len(arr) <= 1:
        return arr
    mid = len(arr) // 2
    left = mergesort(arr[:mid])
    right = mergesort(arr[mid:])
    return merge(left, right)

def merge(left, right):
    sorted_arr = []
    while left and right:
        if left[0] < right[0]:
            sorted_arr.append(left.pop(0))
        else:
            sorted_arr.append(right.pop(0))
    sorted_arr.extend(left)
    sorted_arr.extend(right)
    return sorted_arr

import sys
def prim(graph, n):
    selected = [False] * n
    selected[0] = True
    edges = []
    for _ in range(n - 1):
        min_weight = sys.maxsize
        u, v = -1, -1
        for i in range(n):
            if selected[i]:
                for j in range(n):
                    if not selected[j] and graph[i][j] > 0 and graph[i][j] < min_weight:
                        min_weight = graph[i][j]
                        u, v = i, j
        selected[v] = True
        edges.append((u, v, min_weight))
    print("Minimum Spanning Tree:")
    for u, v, weight in edges:
        print(f"{u} - {v} : {weight}")

10. Write a program to find Minimum cost spanning tree of a given undirected graph using 
Kruskal’s algorithm. 

def find(parent, i):
    if parent[i] != i:
        parent[i] = find(parent, parent[i])  
    return parent[i]

def union(parent, u, v):
    root_u = find(parent, u)
    root_v = find(parent, v)
    parent[root_u] = root_v  

def kruskal(n, graph):
    edges = []
    for i in range(n):
        for j in range(i + 1, n):  
            if graph[i][j] != 0:  
                edges.append((graph[i][j], i, j))  

    edges.sort()

    parent = [i for i in range(n)]  
    mst = []  
    min_cost = 0

    for weight, u, v in edges:
        if find(parent, u) != find(parent, v):  
            union(parent, u, v)  
            mst.append((u, v, weight))
            min_cost += weight

        if len(mst) == n - 1:  
            break

    print("Edges in the constructed MST:")
    for u, v, weight in mst:
        print(f"{u} -- {v} == {weight}")
    print(f"Minimum Spanning Tree Cost: {min_cost}")

11. From a given vertex in a weighted connected graph, find shortest paths to other vertices using Dijkstra's algorithm.
def dijkstra(graph, src, n):
    inf = 99999
    dist = [inf] * n
    dist[src] = 0
    visited = [False] * n
    for _ in range(n):
        u = -1
        for v in range(n):
            if not visited[v] and (u == -1 or dist[v] < dist[u]):
                u = v
        visited[u] = True
        for v in range(n):
            if not visited[v] and graph[u][v] > 0 and dist[u] + graph[u][v] < dist[v]:
                dist[v] = dist[u] + graph[u][v]
    return dist

def knapSack(W, wt, pft, n):
    if n == 0 or W == 0:
        return 0
    if (wt[n-1] > W):
        return knapSack(W, wt, pft, n-1)
    else:
        return max(pft[n-1] + knapSack(W-wt[n-1], wt, pft, n-1),
                   knapSack(W, wt, pft, n-1))
